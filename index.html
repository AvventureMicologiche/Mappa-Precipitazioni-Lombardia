// ═══════════════════════════════════════════════════════════════ // DATA SOURCE: ARPA LOMBARDIA — query giornaliere + cache + progresso // ═══════════════════════════════════════════════════════════════ function setBadgeProgress(done,total){ document.getElementById('badge').textContent='\u23f3 '+done+'/'+total+' giorni...'; } function loadARPALombardia(pastDate,nowDate,labelPeriodo){ var BASE='/.netlify/functions/arpa'; function getDays(from,to){ var days=[]; var d=new Date(from.getFullYear(),from.getMonth(),from.getDate()); var end=new Date(to.getFullYear(),to.getMonth(),to.getDate()); while(d<=end){days.push(new Date(d));d.setDate(d.getDate()+1);} return days; } var days=getDays(pastDate,nowDate); var totalDays=days.length; var stazioni=ARPA_STATIONS||LOMBARDIA_STATIONS; var stMap={}; stazioni.forEach(function(s){ if(s.id) stMap[s.id]={n:s.n,lat:s.lat,lon:s.lon,q:s.q,p:s.p}; }); if(Object.keys(stMap).length<5&&ARPA_STATIONS&&ARPA_STATIONS[0]&&ARPA_STATIONS[0].idsensore){ ARPA_STATIONS.forEach(function(s){ stMap[s.idsensore]={n:s.n||s.nomestazione,lat:s.lat,lon:s.lon,q:s.q||s.quota||0,p:s.p||s.provincia||'\u2014'}; }); } var needFetch=Object.keys(stMap).length<5; function queryDay(sm,day){ var ids=Object.keys(sm); var yyyy=day.getFullYear(); var mm=String(day.getMonth()+1).padStart(2,'0'); var dd=String(day.getDate()).padStart(2,'0'); var dateStr=yyyy+'-'+mm+'-'+dd; var todayStr=new Date().toISOString().substring(0,10); var isToday=(dateStr===todayStr); var cached={}; var toFetch=[]; ids.forEach(function(id){ var k=CACHE_VER+dateStr+'_'+id; try{ var raw=localStorage.getItem(k); if(raw){ var obj=JSON.parse(raw); if(!isToday||(Date.now()-obj.ts)<3600000){cached[id]=obj.tot;return;} } }catch(e){} toFetch.push(id); }); if(toFetch.length===0) return Promise.resolve(cached); var startISO=dateStr+'T00:00:00'; var endISO=dateStr+'T23:59:59'; var mid=Math.ceil(toFetch.length/2); var batches=[toFetch.slice(0,mid),toFetch.slice(mid)]; var fetches=batches.filter(function(b){return b.length>0;}).map(function(batchIds){ var orClause=batchIds.map(function(id){return "idsensore='"+id+"'";}).join(' OR '); var whereStr="data between '"+startISO+"' and '"+endISO+"' AND valore >= '0' AND ("+orClause+')'; var url=BASE+'?endpoint=647i-nhxk.json' +'&$select=idsensore,max(valore)%20as%20mx,min(valore)%20as%20mn' +'&$where='+encodeURIComponent(whereStr) +'&$group=idsensore&$limit=5000'; return fetch(url).then(function(r){ if(!r.ok) throw new Error('HTTP '+r.status); return r.json(); }); }); return Promise.all(fetches).then(function(results){ var dayMap=Object.assign({},cached); results.reduce(function(a,b){return a.concat(b);},[]).forEach(function(pr){ var mx=parseFloat(pr.mx)||0; var mn=parseFloat(pr.mn)||0; var tot=Math.max(0,mx-mn); dayMap[pr.idsensore]=tot; try{ localStorage.setItem(CACHE_VER+dateStr+'_'+pr.idsensore,JSON.stringify({tot:tot,ts:Date.now()})); }catch(e){} }); return dayMap; }); } function doRenderFinal(sm,rainMap){ var allSt=Object.keys(sm).map(function(id){ var st=sm[id]; var mm=rainMap[id]||0; return {n:st.n,lat:st.lat,lon:st.lon,q:st.q,p:st.p, mm:mm<0.5?0:Math.round(mm*10)/10}; }); var out=selectUniform(allSt,200); if(out.length<5) throw new Error('pochi dati'); console.log('[ARPA] OK:',allSt.length,'->',out.length,'stazioni'); render(out,'\uD83D\uDFE2 ARPA live',labelPeriodo); } function loadAllDays(sm){ var rainMap={}; var doneDays=0; var WINDOW=10; setBadgeProgress(0,totalDays); function loadWindow(startIdx){ if(startIdx>=days.length){ doRenderFinal(sm,rainMap); return; } var windowDays=days.slice(startIdx,startIdx+WINDOW); Promise.all(windowDays.map(function(day){return queryDay(sm,day);})) .then(function(results){ results.forEach(function(dayMap){ Object.keys(dayMap).forEach(function(id){ rainMap[id]=(rainMap[id]||0)+(dayMap[id]||0); }); }); doneDays+=windowDays.length; setBadgeProgress(Math.min(doneDays,totalDays),totalDays); if(startIdx+WINDOW=5) render(partialOut,'\u23f3 ARPA caricamento...',labelPeriodo); } loadWindow(startIdx+WINDOW); }) .catch(function(e){ console.warn('[ARPA] errore finestra:',e.message); doneDays+=windowDays.length; loadWindow(startIdx+WINDOW); }); } loadWindow(0); } if(needFetch){ var stUrl=BASE+'?endpoint=nf78-nj6b.json&$limit=500&$select=idsensore,nomestazione,lat,lng,quota,provincia&$where=tipologia=%27Precipitazione%27'; fetch(stUrl) .then(function(r){if(!r.ok)throw new Error('HTTP '+r.status);return r.json();}) .then(function(data){ var sm={}; data.forEach(function(s){ var lat=parseFloat(s.lat),lon=parseFloat(s.lng); if(!isNaN(lat)&&!isNaN(lon)&&lat>44&&lat<47&&lon>7&&lon<12) sm[s.idsensore]={n:s.nomestazione||'\u2014',lat:lat,lon:lon,q:s.quota||0,p:s.provincia||'\u2014'}; }); var selected=selectUniform(Object.values(sm).map(function(v,i){v.idsensore=Object.keys(sm)[i];return v;}),200); var sm2={}; selected.forEach(function(s){sm2[s.idsensore]=s;}); ARPA_STATIONS=selected; loadAllDays(sm2); }) .catch(function(e){ console.warn('[ARPA] fallback Open-Meteo:',e.message); loadOpenMeteo(ARPA_STATIONS||LOMBARDIA_STATIONS,pastDate,nowDate,labelPeriodo,'\uD83C\uDF0D'); }); } else { try{loadAllDays(stMap);} catch(e){ console.warn('[ARPA] fallback Open-Meteo:',e.message); loadOpenMeteo(ARPA_STATIONS||LOMBARDIA_STATIONS,pastDate,nowDate,labelPeriodo,'\uD83C\uDF0D'); } } }
